### Tool 1: Batch Translation Script (Google Translate - Free)

```javascript
// scripts/batchTranslate.js
// Uses free Google Translate (no API key needed)

const translate = require('google-translate-api-x');
const fs = require('fs');
const phrases = require('../src/data/phrases');

async function batchTranslate(targetLang) {
  const results = [];

  for (const phrase of phrases) {
    try {
      const res = await translate(phrase.english, { to: targetLang });

      results.push({
        id: phrase.id,
        english: phrase.english,
        current: phrase[targetLang],
        google: res.text,
        match: phrase[targetLang] === res.text
      });

      // Rate limit: wait 100ms between requests
      await new Promise(resolve => setTimeout(resolve, 100));
    } catch (error) {
      console.error(`Error translating phrase ${phrase.id}:`, error);
    }
  }

  // Save results
  fs.writeFileSync(
    `reviews/${targetLang}_google.json`,
    JSON.stringify(results, null, 2)
  );

  // Summary
  const matches = results.filter(r => r.match).length;
  console.log(`${targetLang}: ${matches}/${results.length} match (${(matches/results.length*100).toFixed(1)}%)`);
}

// Usage
batchTranslate('ja'); // Japanese
batchTranslate('ko'); // Korean
// etc.
```

### Tool 2: Comparison Script (Multiple Services)

```javascript
// scripts/compareTranslations.js
const translate = require('google-translate-api-x');
const axios = require('axios');

async function compareTranslations(english, targetLang) {
  const results = {
    english: english,
    targetLang: targetLang,
    services: {}
  };

  // Google Translate (free)
  try {
    const google = await translate(english, { to: targetLang });
    results.services.google = google.text;
  } catch (e) {
    results.services.google = 'ERROR';
  }

  // Microsoft Translator (free)
  try {
    const microsoft = await axios.get(
      `https://api.mymemory.translated.net/get?q=${encodeURIComponent(english)}&langpair=en|${targetLang}`
    );
    results.services.microsoft = microsoft.data.responseData.translatedText;
  } catch (e) {
    results.services.microsoft = 'ERROR';
  }

  // Check consensus
  const translations = Object.values(results.services);
  const unique = [...new Set(translations)];

  results.consensus = unique.length === 1;
  results.needsReview = unique.length > 2;

  return results;
}

// Usage
compareTranslations('Where is the bathroom?', 'ja').then(console.log);
```

```javascript
// scripts/generateReviewCSV.js
const fs = require('fs');

function generateReviewCSV(language, comparisonResults) {
  const csv = [
    'ID,Category,English,Current,Google,DeepL,Microsoft,Status,Corrected,Notes'
  ];

  for (const result of comparisonResults) {
    csv.push([
      result.id,
      result.categoryId,
      `"${result.english}"`,
      `"${result.current}"`,
      `"${result.google || ''}"`,
      `"${result.deepl || ''}"`,
      `"${result.microsoft || ''}"`,
      result.consensus ? 'OK' : 'REVIEW',
      '', // Empty for manual correction
      '' // Empty for notes
    ].join(','));
  }

  fs.writeFileSync(
    `reviews/${language}_review.csv`,
    csv.join('\n')
  );

  console.log(`Generated: reviews/${language}_review.csv`);
}
```
